query EventStandingsQuery($slug: String!, $numEntrants: Int) {
	event(slug: $slug) {
    tournament {
      name
    }
    id
    startAt
    numEntrants
    standings (query : {
      perPage: $numEntrants,
      page: 1
    }){
      nodes {
        placement
        entrant {
          id
          name
          participants {
            user {
              slug
            }
          }
        }
      }
    }
  }
}
