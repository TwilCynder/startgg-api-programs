query User($slug: String, $id: ID, $perPage: Int, $page: Int!, $after: Timestamp) {
    user(slug: $slug, id: $id){
        slug
        id
        genderPronoun
        location {
            city
            country
            state
        }
        player {
            sets(perPage: $perPage, page: $page, filters : {
                updatedAfter: $after
            }) {
                nodes {
                    event {
                        slug
                    }
                    completedAt
                    slots {
                        entrant {
                            id
                            participants {
                                user {
                                    id
                                } 
                                player {
                                    id
                                }
                            } 
                        }
                        standing {
                            placement
                        }
                    }
                    games {
                        selections {
                            entrant {
                                id
                            }
                            selectionValue
                        }
                    }         
                }
            }
        }
    }
}
