query EventSetsQuery($slug: String!, $page: Int, $perPage: Int) {
    event(slug: $slug) {
        sets  (page: $page, perPage: $perPage)  {
            nodes {
                slots{
                    entrant {
                        name
                        participants {
                            player {
                                user {
                                    id
                                }
                                gamerTag
                            }
                        }
                        seeds {
                            seedNum
                        }
                    }
                  	standing {
                      stats {
                        score {
                          value
                        }
                      }
                    }
                }
            }
        }
    }
}
