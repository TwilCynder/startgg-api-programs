query StandinsgQuery($id: ID, $slug: String, $standingsPage: Int, $stadingsPerPage: Int, $eventsPage: Int, $eventsPerPage: Int) {
	user(id: $id, slug: $slug) {
    events (query : {
      page: $eventsPage,
      sortBy: "startAt desc",
      perPage: $eventsPerPage
    }){
      nodes {
        tournament {
          name
        }
        entrantSizeMin
        id
        startAt
        standings (query : {
          perPage: $stadingsPerPage,
          page: $standingsPage
        }){
          nodes {
            placement
            entrant {
              participants {
                user {
                  id
                }
              }
            }
          }
        }
      }
    }
    
  }	
}
