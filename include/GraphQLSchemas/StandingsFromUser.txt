query StandinsgQuery($id: ID, $slug: String, $page: Int!) {
	user(id: $id, slug: $slug) {
    events (query : {
      perPage: 3,
      page: $page,
      sortBy: "startAt desc"
    }){
      nodes {
        tournament {
          name
        }
        entrantSizeMin
        id
        startAt
        standings (query : {
          perPage: 48,
          page: 1
        }){
          nodes {
            placement
            entrant {
              participants {
                user {
                  id
                }
              }
            }
          }
        }
      }
    }
    
  }	
}
