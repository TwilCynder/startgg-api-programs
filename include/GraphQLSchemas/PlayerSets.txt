query PlayerSets($id: ID!, $perPage: Int!, $page: Int!, $after: Timestamp) {
    player (id: $id) {
        sets(perPage: $perPage, page: $page, filters : {
          updatedAfter: $after
        }) {
            nodes {
                completedAt
                slots {
                    entrant {
                        participants {
                            player {
                                id
                                gamerTag
                            }
                            user {
                                id
                            }    
                        } 
                    }
                    standing {
                        placement
                    }
                }
                displayScore         
            }
        }
    }
}
