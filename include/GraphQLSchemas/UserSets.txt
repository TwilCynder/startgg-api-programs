query User($slug: String, $id: ID, $perPage: Int, $page: Int!, $after: Timestamp) {
    user(slug: $slug, id: $id){
        player {
            sets(perPage: $perPage, page: $page, filters : {
            updatedAfter: $after
            }) {
                nodes {
                    completedAt
                    slots {
                        entrant {
                            participants {
                                player {
                                    id
                                    gamerTag
                                }
                                user {
                                    id
                                }  
                            } 
                        }
                        standing {
                            placement
                        }
                    }
                    displayScore         
                }
            }
        }
    }
}
